--1)Создание таблицы

CREATE TABLE Продукты (
id integer primary key autoincrement,
Название varchar(255),
Цена float,
Количество integer,
Комментарий varchar(255) 
)

--2)Чтение
SELECT id FROM Продукты;

UPDATE Продукты

DELETE FROM Продукты
